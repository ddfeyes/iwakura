# Spec: PSYCHE — Recent Decisions + Tasks from STATE.yaml + Memory (Issue #46)

## Problem
PSYCHE screen shows mood + basic state but lacks rich context: what Lain is working on, recent decisions, memory activity.

## Solution

### Backend (backend/main.py) — enhance /api/psyche

Add helper functions:

```python
import re
from pathlib import Path
from datetime import datetime

LAIN_BASE = Path.home() / "agents" / "lain"
STATE_FILE = LAIN_BASE / "STATE.yaml"
MEMORY_DIR = LAIN_BASE / "memory"

def get_recent_decisions() -> list[str]:
    """Extract last 5 decisions/events from today's diary file."""
    today = datetime.utcnow().strftime("%Y-%m-%d")
    diary = MEMORY_DIR / f"{today}.md"
    if not diary.exists():
        # Try yesterday
        from datetime import timedelta
        yesterday = (datetime.utcnow() - timedelta(days=1)).strftime("%Y-%m-%d")
        diary = MEMORY_DIR / f"{yesterday}.md"
    if not diary.exists():
        return []
    try:
        text = diary.read_text(errors="replace")
        # Extract H2/H3 headings and bullet points
        lines = text.splitlines()
        entries = []
        for line in lines:
            line = line.strip()
            if line.startswith("## ") or line.startswith("### "):
                entries.append(line.lstrip("#").strip())
            elif line.startswith("- ") and len(line) > 10:
                entries.append(line[2:].strip()[:100])
        return entries[-8:]  # last 8 items
    except Exception:
        return []

def get_active_task() -> dict:
    """Read current task from STATE.yaml."""
    if not STATE_FILE.exists():
        return {}
    try:
        import yaml  # may not be installed
        with open(STATE_FILE) as f:
            data = yaml.safe_load(f)
        return {
            "goal": str(data.get("goal", data.get("task", ""))),
            "status": str(data.get("status", "")),
            "remaining": data.get("remaining", []),
        }
    except ImportError:
        # Fall back to simple text parse
        text = STATE_FILE.read_text(errors="replace")
        task = {}
        for line in text.splitlines():
            if line.startswith("goal:") or line.startswith("task:"):
                task["goal"] = line.split(":", 1)[1].strip().strip('"')
            elif line.startswith("status:"):
                task["status"] = line.split(":", 1)[1].strip().strip('"')
        return task
    except Exception:
        return {}

def get_memory_activity() -> dict:
    """Count memory files, get last modified time."""
    try:
        files = list(MEMORY_DIR.glob("*.md"))
        if not files:
            return {}
        latest = max(files, key=lambda f: f.stat().st_mtime)
        return {
            "file_count": len(files),
            "latest_file": latest.name,
            "latest_mtime": datetime.fromtimestamp(latest.stat().st_mtime).strftime("%Y-%m-%d %H:%M"),
        }
    except Exception:
        return {}
```

Update `/api/psyche` to include these in the response:
```python
return {
    ...existing fields...,
    "recent_decisions": get_recent_decisions(),
    "active_task": get_active_task(),
    "memory_activity": get_memory_activity(),
}
```
