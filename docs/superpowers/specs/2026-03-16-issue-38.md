# Spec: Diary UX Polish — Auto-scroll, Unread Badge, Typing Indicator (Issue #38)

**Date:** 2026-03-16  
**Issue:** #38  
**Priority:** P1

## Changes

### `frontend/js/diary.js`

**Auto-scroll:**
```js
let userScrolledUp = false;

// In scroll listener on messages container:
messagesEl.addEventListener('scroll', () => {
    const atBottom = messagesEl.scrollHeight - messagesEl.scrollTop - messagesEl.clientHeight < 100;
    userScrolledUp = !atBottom;
});

// After appending any message row:
function scrollToBottom() {
    if (!userScrolledUp) {
        messagesEl.scrollTop = messagesEl.scrollHeight;
    }
}
// Reset userScrolledUp to false when user sends a message (they want to see the response)
```

**Typing indicator:**
```js
let typingEl = null;

function showTyping() {
    if (typingEl) return;
    typingEl = document.createElement('div');
    typingEl.className = 'message lain-message typing-indicator';
    typingEl.innerHTML = '<span class="typing-dots"><span>.</span><span>.</span><span>.</span></span>';
    messagesEl.appendChild(typingEl);
    scrollToBottom();
}

function hideTyping() {
    if (typingEl) { typingEl.remove(); typingEl = null; }
}

// Call showTyping() after sending message
// Call hideTyping() when receiving lain message or error
```

**Unread badge:**
```js
let unreadCount = 0;
let isDiaryActive = false;  // set true when diary screen shown, false when leaving

function incrementUnread() {
    if (isDiaryActive) return;
    unreadCount++;
    updateBadge();
}

function clearUnread() {
    unreadCount = 0;
    updateBadge();
}

function updateBadge() {
    const badge = document.getElementById('diary-unread-badge');
    if (!badge) return;
    badge.textContent = unreadCount > 0 ? String(unreadCount) : '';
    badge.style.display = unreadCount > 0 ? 'flex' : 'none';
}
```

In app.js / nav.js: when screen changes to 'diary', call `window.iwakura.clearDiaryUnread?.()`.
Export: `window.iwakura.clearDiaryUnread = clearUnread; window.iwakura.setDiaryActive = (v) => isDiaryActive = v;`
